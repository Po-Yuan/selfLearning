//after improving
class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        set<string> st;
        for(auto &w :emails){
            string temp = "";
            for(int i= 0 ; i!= w.size() ; i++){
                if(w[i] == '.') continue;
                if(w[i] == '@'){
                    temp += w.substr(i,w.size());
                    cout<<temp<<endl;
                    break;} 
                if(w[i] == '+'){
                    while(w[++i] != '@');
                    cout<<temp + "-1"<<endl;
                    temp += w.substr(i,w.size());
                    cout<<temp + "-2"<<endl;
                    
                    break;
                    
                }
                cout<<i<<endl;
                temp.push_back(w[i]);
            }
            st.insert(temp);
        }
        return st.size();
    }
};

//set為使用key方式儲存，其key即為value並且在set中是唯一值，並且無法被改變。其可以被插入或者移除，並且可以使用迭代器(iterator)取出並且排序。在此改進方式用set的值唯一性特性進行儲存資料判斷是否重複mail。
