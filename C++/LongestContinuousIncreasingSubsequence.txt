class Solution {
public:
int findLengthOfLCIS(vector<int>& nums) {
    int result = 1;
    int cint = 1;
    
    for(int i = 1; i !=nums.size(); i++){
        if(nums[i]>nums[i-1]){
            cint++;
        }else{
            cint = 1;
        }
        result = max(result,cint);
    }
    
    return result;
}
};
