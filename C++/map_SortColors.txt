class Solution {
public:
    void sortColors(vector<int>& nums) {
        map<int,int> map;
        int index=0 ;
        int len;
        for(int i =0;i!=nums.size();i++){
            map[nums[i]]++;
        }
        
        
        
        for(auto m :map){
            cout<<m.first<<":"<<m.second<<endl;
            cout<<index<<endl;
            
            for(int i=0; i!= m.second ; i++){
                nums[index] = m.first;
                index++;
            }
        }
    }
};
